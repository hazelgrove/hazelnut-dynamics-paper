  \tarr{\htau_1}{\htau_2} \neq \tarr{\htau_3}{\htau_4}\\

\inferrule[IEHole]
{ }

 \isIndet{\dexp_1}\\
% \isFinal{\dexp_2}~\text{\cy{??}}}

{\isIndet{\dap{\dexp_1}{\dexp_2}}}

}

\inferrule[ICastHoleGround] {

  \isGround{\htau}
